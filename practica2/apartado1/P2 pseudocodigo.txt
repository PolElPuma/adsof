Función importeMensual(lector)
    total = 0
    plan = verPlan(lector)

    if plan es igual a 0 entonces
        total += precioLibros(lector)
        total += precioSagas(lector)
        total += precioActos(lector)
    else if plan es igual a 1 entonces
        total += precioLibros(lector)
        total += precioSagas(lector)
        total += precioActos(lector)
        total /= 2
        total += 9.99
    else
        total = 19.99

    return total

Función precioLibros(lector)
	libros[] = verLibros(lector)
	total = 0
	for libro in libros:
		total += verPrecio(libro)
	return total

Función precioSagas(lector)
	sagas[] = verSagas(lector)
	total = 0
	for saga in sagas:
		total += verPrecio(saga)
	return total

Función precioActos(lector)
	actos[] = verActos(lector)
	total = 0
	for acto in actos:
		if(descuento = verDescuento(acto))
			total += verPrecio(acto) * descuento
		else
			total += verPrecio(acto)
		
	return total
